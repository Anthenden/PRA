\section{Etude des triplets de Darboux}



\begin{frame}



\begin{Prop}
Soit $E$ un espace de dimension 2 et $L_1 : E \to \R^3$ une application linéaire injective. Soit $L_2 : E \to \R^3$ une application linéaire telle que 
	\[\forall u \in E,  L_1(u) \cdot L_2(u) =0
\]
 Alors il existe un unique $h \in \R^3$ tel que : 
\[\label{defo_infin} L_2=h \times L_1 \]
\end{Prop}
\end{frame}
	


\section{Structure pseudo-conforme d'une variété}



\subsection{Structure pseudo-conforme projective sur une variété}

\begin{frame}

\begin{Def}
La structure pseudo-conforme projective de $M$ en $p$ est l'ensemble des applications bilinéaires $\ff_p \circ \varphi$, où $\varphi : N_pM \to \R$ est un isomorphisme (linéaire). \\
\end{Def} 
\pause
Si $M=\{ (x,t) \mid t=q(x)\}$ alors la structure pseudo-conforme projective de $M$ est $\{(0,-2\lambda b) \mid \lambda \neq 0\}$ en tout point de $M$.


\end{frame}

\subsection{Compactification}
\begin{frame}
Soit $q$ une forme quadratique sur $\R^n$, non dégénérée de signature $(r,n-r)$.
\begin{itemize}
\item $\R^n \simeq M=\{ (x,t) \mid t=q(x)\}$.
\pause
\item On ajoute les points à l'infini 
\pause
\item On obtient une quadrique projective lisse de signature $(r+1,n-r+1)$ dans $\R P^{n+1}$
\pause
\item Cette compactification est conforme.
\end{itemize}



\end{frame}


\section{Géométrie : Etude de $\varphi_{\pm}$}

\subsection{ Géométrie de $Q$ }


\begin{frame}
\begin{Thm}
Tout sous-espace totalement isotrope peut se décrire comme le graphe d'une application linéaire orthogonale
\end{Thm}
\pause
En particulier, les SETIM sont les graphes des éléments de $O(4)$. \end{frame}
\begin{frame}
\begin{Thm}
$SO(4) \simeq S^3 \times S^3/\{\pm(1,1)\}$ (en tant que groupes de Lie)  
\end{Thm}
\pause
\begin{itemize}
    \item $SO(4)$ (et $O^-(4)$) s'identifie à $Q$ (comme variété différentiable). 
    \pause
    \item  $Q$ contient deux familles $Q_+$et $Q_-$ de SETIM
    \pause
    \item ... qui sont elles-mêmes deux copies de $Q$.
\end{itemize}

 
  
 \end{frame}
\begin{frame}
Structure d'incidence entre $Q,Q_+,Q_-$ : 
\pause
\begin{itemize}
    \item $x \in Q$ et $F \in Q_\pm$ sont incidents si $x \in F$
    \pause
    \item $F$ et $G$ sont incidents si $F \cap G$ est de dimension maximale
\end{itemize}
\pause
\begin{Prop} 
Soit $V$ un sous-espace totalement isotrope de dimension $3$ de $\R^4 \oplus \R^4$. Il existe exactement deux sous-espaces totalement isotropes maximaux contenant $V$.
\end{Prop}
\end{frame}
\subsection{Les applications $\varphi_\pm$ associées à un triplet de Darboux}

\begin{frame}
\begin{itemize}
    \item $g$ est une déformation infinitésimale isométrique de $f$ si, et seulement si, $\tau_p \varphi:=P( \varphi(p) \oplus Im(d_p \psi))$ est totalement isotrope 
    \pause
    \item Dans ce cas, $\tau_p \varphi$ est contenu dans exactement deux espaces projectifs de dimension 3, un dans $Q_+$ et l'autre dans $Q_-$. 
    \pause
    \item On peut définir deux applications $\varphi_\pm : S \to Q_\pm$ qui associe à $p$ l'élément de $Q_\pm$ contenant $\tau_p \varphi$. \\
\end{itemize}
\end{frame}


\subsubsection{Calcul de $\varphi_{\pm}$}
\begin{frame}
\begin{itemize}
    \item $\varphi_+=(h,\widetilde{g})_+$
    
    \item $\varphi_-=(\widetilde{h},f^*)_-$
\end{itemize}
\vspace{1cm}

où $f^*=-\frac{v_f}{v_f \cdot f}$ et $\widetilde{h}=h-f^* \times \widetilde{g}$ et \\
$(a,b)_+=\{ [x : s: y :t] \mid y=a \times x+bs, t=-b \cdot x \}$\\
$(a,b)_-=\{ [x : s: y :t] \mid y=a \times x+bt, s=-b \cdot x \}$ 

\end{frame}

\section{Octonions déployés}


\subsection{Généralités}
\begin{frame}
\begin{itemize}
    \item L'algèbre des octonions déployés $\Ocal$ est l'espace vectoriel $\R^8=\H \oplus \H$ muni du produit : \\
\[(a,b)\times (c,d)=(ac+db^*,a^*d+cb)\]
\pause
\item Cette algèbre n'est ni commutative ni associative.
\pause
\item $(a,b)^*=(a^*,-b)$. 
\pause
\item $N : (a,b) \in \Ocal \mapsto (a,b)(a,b)^*=(|a|^2-|b|^2,0) \in \R$ est multiplicative et de signature $(4,4)$. 
\end{itemize}
\end{frame}


\subsection{Symétries entre $Q_+,Q_-$ et $Q$}


\begin{frame}



\begin{Thm}
Un espace vectoriel $F$ est un SETIM si, et seulement si, il peut s'écrire comme l'annulateur à gauche (ou à droite) d'un octonion de norme nulle.  
\end{Thm}
\pause
\begin{itemize}
    \item $Q_+$ est l'ensemble des annulateurs à gauche
    \pause
    \item $Q_-$ est l'ensemble des annulateurs à droite 
\end{itemize}



\end{frame}


\begin{frame}
\begin{Thm} 
Soient $L_a=\{x \in \Ocal \mid ax=0\}$ et $R_b=\{x \in \Ocal \mid  xb=0\}$ deux SETIM. Leur intersection est maximale (i.e. de dimension 3)  si, et seulement si, $ba=0$. 
Les éléments de cette intersection sont alors de la forme $a^*y$ où $\left\langle y,b^*\right\rangle=0$
\end{Thm}
\end{frame}
\begin{frame}
\begin{Thm}
Le groupe des bijections de l'union disjointe $Q\cup Q_+ \cup Q_-$ qui respectent les relations d'incidence agit sur les composantes en les permutant arbitrairement.
\end{Thm}
\end{frame}
\begin{frame}
L'algèbre $\Zcal$ des matrices de Zorn : \\
L'espace vectoriel des matrices $\begin{pmatrix} a & x \\ y & b \end{pmatrix}$ où  $a,b \in \R, x,y \in \R^3$ muni du produit : 
\[\begin{pmatrix} a & x \\ y & b \end{pmatrix} \times \begin{pmatrix} a' & x' \\ y' & b' \end{pmatrix}=\begin{pmatrix} aa'+x \cdot y' & ax'+b'x-y \times y' \\ ay'+by'+x \times x' & bb'+x' \cdot y \end{pmatrix}.
	\] 
Conjugaison dans $\Zcal$ : $\overline{\begin{pmatrix} a & x \\ y & b \end{pmatrix}}=\begin{pmatrix} b & -x \\ -y & a \end{pmatrix}$
\end{frame}

\subsection{Trialité différentielle}


\begin{frame}

\begin{Thm}

	\[ \psi_+\psi =0 ; \psi_+ d\psi=0\]
	\pause
\[ \psi\psi_- =0 ;  (d\psi) \psi_-=0\]
\pause
	\[ \psi_-\psi_+ =0 \]
	\pause
	\[(d\psi_+)\psi =0 \]
	\pause
	\[(d \psi_-) \psi_+=\psi_- d \psi_+=0\]
\end{Thm}
\end{frame}

\begin{frame}
\begin{Prop}\label{composition}
Si $\varphi : S \to Q$ est une immersion totalement isotrope telle que $\varphi_+$ soit une immersion, on a $\varphi_{++}=\varphi_-$ et $\varphi_{+-}=\varphi$. De même, si $\varphi_-$ est une immersion, on a $\varphi_{--}=\varphi_+$ et $\varphi_{-+}=\varphi$
\end{Prop}
\begin{itemize}
  \item si $c$ est un anti-automorphisme, $(c \circ \varphi)_+=c \circ \varphi_-$ \\ 
  \item si $\sigma$ est un automorphisme, $(\sigma \circ \varphi)_+=\sigma \circ \varphi_+$.
  \end{itemize}
  
\end{frame}
